// ========== ຕົວປ່ຽນປ້ອງກັນການສົ່ງຟອມຊໍ້າ ========== let isSubmitting = false; // ຟັງຊັນຊ່ວຍສຳລັບການສົ່ງຟອມ async function handleSubmit(e, saveFunction) { e.preventDefault(); if (isSubmitting) { console.log('ກຳລັງດຳເນີນການ, ກະລຸນາລໍຖ້າ...'); return; } isSubmitting = true; const submitBtn = e.target.querySelector('button[type="submit"]'); if (submitBtn) { submitBtn.disabled = true; submitBtn.textContent = 'ກຳລັງບັນທຶກ...'; } try { await saveFunction(); } catch (err) { console.error(err); alert('ເກີດຂໍ້ຜິດພາດ: ' + err.message); } finally { isSubmitting = false; if (submitBtn) { submitBtn.disabled = false; submitBtn.textContent = 'ບັນທຶກ'; } } } // ລຶບ event listeners ເກົ່າ (ຖ້າມີ) ໂດຍການສ້າງຟອມຂຶ້ນໃໝ່ // ແກ້ໄຂຟອມສິນຄ້າ const productForm = document.getElementById('productForm'); const newProductForm = productForm.cloneNode(true); productForm.parentNode.replaceChild(newProductForm, productForm); newProductForm.addEventListener('submit', (e) => handleSubmit(e, async () => { const id = document.getElementById('prodId').value; const data = { name: document.getElementById('prodName').value, price: parseFloat(document.getElementById('prodPrice').value) || 0, category: document.getElementById('prodCategory').value, image: document.getElementById('prodImage').value, status: document.getElementById('prodStatus').value, updatedAt: serverTimestamp() }; if (id) { await updateDoc(doc(db, 'products', id), data); } else { data.createdAt = serverTimestamp(); await addDoc(collection(db, 'products'), data); } closeModal('productModal'); loadProducts(); loadDashboard(); })); // ແກ້ໄຂຟອມບໍລິການ const serviceForm = document.getElementById('serviceForm'); const newServiceForm = serviceForm.cloneNode(true); serviceForm.parentNode.replaceChild(newServiceForm, serviceForm); newServiceForm.addEventListener('submit', (e) => handleSubmit(e, async () => { const id = document.getElementById('svcId').value; const data = { name: document.getElementById('svcName').value, category: document.getElementById('svcCategory').value, description: document.getElementById('svcDesc').value, imageUrl: document.getElementById('svcImage').value, status: 'active', updatedAt: serverTimestamp() }; if (id) { await updateDoc(doc(db, 'services', id), data); } else { data.createdAt = serverTimestamp(); await addDoc(collection(db, 'services'), data); } closeModal('serviceModal'); loadServices(); })); // ແກ້ໄຂຟອມຜົນງານ const portfolioForm = document.getElementById('portfolioForm'); const newPortfolioForm = portfolioForm.cloneNode(true); portfolioForm.parentNode.replaceChild(newPortfolioForm, portfolioForm); newPortfolioForm.addEventListener('submit', (e) => handleSubmit(e, async () => { const id = document.getElementById('portId').value; const data = { title: document.getElementById('portTitle').value, category: document.getElementById('portCategory').value, description: document.getElementById('portDesc').value, imageUrl: document.getElementById('portImage').value, status: 'active', updatedAt: serverTimestamp() }; if (id) { await updateDoc(doc(db, 'portfolio', id), data); } else { data.createdAt = serverTimestamp(); await addDoc(collection(db, 'portfolio'), data); } closeModal('portfolioModal'); loadPortfolio(); })); // ແກ້ໄຂຟອມແບຣນ const brandForm = document.getElementById('brandForm'); const newBrandForm = brandForm.cloneNode(true); brandForm.parentNode.replaceChild(newBrandForm, brandForm); newBrandForm.addEventListener('submit', (e) => handleSubmit(e, async () => { const id = document.getElementById('brandId').value; const data = { name: document.getElementById('brandName').value, category: document.getElementById('brandCategory').value, logoUrl: document.getElementById('brandLogo').value, status: 'active', updatedAt: serverTimestamp() }; if (id) { await updateDoc(doc(db, 'brands', id), data); } else { data.createdAt = serverTimestamp(); await addDoc(collection(db, 'brands'), data); } closeModal('brandModal'); loadBrands(); })); // ແກ້ໄຂຟອມຂ່າວ const newsForm = document.getElementById('newsForm'); const newNewsForm = newsForm.cloneNode(true); newsForm.parentNode.replaceChild(newNewsForm, newsForm); newNewsForm.addEventListener('submit', (e) => handleSubmit(e, async () => { await addDoc(collection(db, 'news'), { title: document.getElementById('newsTitle').value, category: document.getElementById('newsCategory').value, content: document.getElementById('newsContent').value, createdAt: serverTimestamp() }); closeModal('newsModal'); loadNews(); })); // ແກ້ໄຂຟອມສະໄລເດີ const sliderForm = document.getElementById('sliderForm'); const newSliderForm = sliderForm.cloneNode(true); sliderForm.parentNode.replaceChild(newSliderForm, sliderForm); newSliderForm.addEventListener('submit', (e) => handleSubmit(e, async () => { await addDoc(collection(db, 'sliders'), { title: document.getElementById('sliderTitle').value, linkUrl: document.getElementById('sliderLink').value, image: document.getElementById('sliderImage').value, status: document.getElementById('sliderStatus').value, createdAt: serverTimestamp() }); closeModal('sliderModal'); loadSliders(); })); // ຟັງຊັນອັບໂຫຼດຮູບຜ່ານ ImgBB (ປັບປຸງໃຫ້ໃຊ້ງານດີຂຶ້ນ) async function uploadImageToImgBB(callback) { const input = document.createElement('input'); input.type = 'file'; input.accept = 'image/*'; input.onchange = async (e) => { const file = e.target.files[0]; if (!file) return; // ກວດສອບຂະໜາດໄຟລ໌ (ບໍ່ເກີນ 5MB) if (file.size > 5 * 1024 * 1024) { alert('ຮູບພາບມີຂະໜາດໃຫຍ່ເກີນໄປ (ສູງສຸດ 5MB)'); return; } const formData = new FormData(); formData.append('image', file); const btn = input.previousSibling; if (btn) { btn.disabled = true; btn.textContent = 'ກຳລັງອັບ...'; } try { const res = await fetch(`https://api.imgbb.com/1/upload?key=85d8acf7f765f021b6c76dfae3650f53`, { method: 'POST', body: formData }); const data = await res.json(); if (data.success) { callback(data.data.url); alert('ອັບໂຫຼດສຳເລັດ!'); } else { alert('ອັບໂຫຼດບໍ່ສຳເລັດ: ' + (data.error?.message || 'ບໍ່ຮູ້ສາເຫດ')); } } catch (err) { console.error(err); alert('ເກີດຂໍ້ຜິດພາດໃນການອັບໂຫຼດ'); } finally { if (btn) { btn.disabled = false; btn.textContent = 'ອັບໂຫຼດຮູບ'; } } }; input.click(); } // ປະກາດຟັງຊັນອັບໂຫຼດຮູບສຳລັບແຕ່ລະປະເພດ window.uploadProductImage = () => uploadImageToImgBB((url) => document.getElementById('prodImage').value = url); window.uploadServiceImage = () => uploadImageToImgBB((url) => document.getElementById('svcImage').value = url); window.uploadPortfolioImage = () => uploadImageToImgBB((url) => document.getElementById('portImage').value = url); window.uploadBrandLogo = () => uploadImageToImgBB((url) => document.getElementById('brandLogo').value = url); window.uploadSliderImage = () => uploadImageToImgBB((url) => document.getElementById('sliderImage').value = url);
